\textbf{Remarks}
\begin{enumerate}
\item If $\pdfXY = g(x)h(y)$, where $g(x)$ and $h(y)$ are pdf's, then $X$ and $Y$ are independent.

\item If $\cdfXY = G(x)H(y)$, where $G(x)$ and $H(y)$ are cdf's, then $X$ and $Y$ are independent.
\vspace{6cm}

\end{enumerate}
